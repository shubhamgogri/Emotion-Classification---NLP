name: Sync to Hugging Face hub

on:
  push:
    branches: [main]
  #to run this workflow manually from the Action tab
  workflow_dispatch:

jobs:
  sync-to-hub:
    runs-on: ubuntu-latest
    steps:
     - uses: actions/checkout@v2
       with:
         fetch-depth: 0 
     - name: Push to hub
       env:
         HF_TOKEN: ${{ secrets.HFTOKEN }}
       run: git push --force https://SoorajAI:$HF_TOKEN@huggingface.co/spaces/SoorajAI/Senti-anaIysis main
